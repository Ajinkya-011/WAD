```bash
title = 'Registration Form';
  displayName = '';
  displayAddress = '';
  displayMobile = '';
  displayEmail = '';

  getValue(name: string, address: string, mobile: string, email: string): void {
    this.displayName = name;
    this.displayAddress = address;
    this.displayMobile = mobile;
    this.displayEmail = email;
  }
```
